 class Operators {
 
  constructor(board) {
    this.board = board;
    this.board.innerHTML = '0';
    this.currentOperator = '';
    this.firstNumber = 0;
    this.secondNumber = 0;
    this.operators = {
      '+': this.addition,
      '-': this.subtraction,
      'รท': this.division,
      'x': this.multiplication
    }
    this.needClear = true;
  }
 
  clearBoard() {
    this.board.innerHTML = '0';
    this.needClear = true;
  }
 
  checkForClear() {
    if (this.needClear) {
      this.board.innerHTML = '';
      this.needClear = false;
    }
  }
 
  operatorButton(event) {
    let newOperator = event.target.innerHTML;
    if (this.currentOperator === '') {
      if (newOperator === '=') {
        return;
      }
      this.currentOperator = newOperator;
      this.firstNumber = parseInt(this.board.innerHTML);
      this.needClear = true;
    } else {
      if (newOperator === this.currentOperator) {
        return;
      }
      this.secondNumber = parseInt(this.board.innerHTML);
      this.operators[this.currentOperator].bind(this)();
      this.currentOperator = '';
      this.firstNumber = parseInt(this.board.innerHTML);
      this.secondNumber = 0;
      if (newOperator !== '=') {
        this.currentOperator = newOperator;
      }
      this.needClear = true;
    }
  }
 
  addition() {
    this.board.innerHTML = this.firstNumber + this.secondNumber;
  }
 
  subtraction() {
    this.board.innerHTML = this.firstNumber - this.secondNumber;
  }
 
  multiplication() {
    this.board.innerHTML = this.firstNumber * this.secondNumber;
  }
 
  division() {
    this.board.innerHTML = this.firstNumber / this.secondNumber;
  }
}
 
export const operators = Operators;
